    stop_on_error "–í–µ—Ä—Å–∏—è –Ω–µ –æ–±–Ω–æ–≤–ª–µ–Ω–∞ –≤ $VERSION_ERRORS —Ñ–∞–π–ª(–∞—Ö):$VERSION_ERROR_FILES"
fi
echo "‚úì –í—Å–µ –≤–µ—Ä—Å–∏–∏ —É—Å–ø–µ—à–Ω–æ –æ–±–Ω–æ–≤–ª–µ–Ω—ã"

# –®–ê–ì 10: –î–æ–±–∞–≤–ª—è–µ–º –∏–∑–º–µ–Ω–µ–Ω–Ω—ã–µ —Ñ–∞–π–ª—ã —è–≤–Ω–æ (–ù–ï –∏—Å–ø–æ–ª—å–∑—É–µ–º git add .)
echo "=== –®–ê–ì 10: –î–æ–±–∞–≤–ª–µ–Ω–∏–µ —Ñ–∞–π–ª–æ–≤ –≤ –∏–Ω–¥–µ–∫—Å ==="
# –ö–†–ò–¢–ò–ß–ù–û: –ò—Å–ø–æ–ª—å–∑—É–µ–º while read –¥–ª—è –æ–±—Ä–∞–±–æ—Ç–∫–∏ –º–Ω–æ–≥–æ—Å—Ç—Ä–æ—á–Ω—ã—Ö –∑–Ω–∞—á–µ–Ω–∏–π (–Ω–µ for, —Ç–∞–∫ –∫–∞–∫ —Ä–∞–∑–¥–µ–ª—è–µ—Ç –ø–æ –ø—Ä–æ–±–µ–ª–∞–º)
# –≠—Ç–æ –∫—Ä–∏—Ç–∏—á–µ—Å–∫–∏ –≤–∞–∂–Ω–æ! –ü–µ—Ä–µ–º–µ–Ω–Ω—ã–µ CHANGED_FILES, NEW_FILES, DELETED_FILES –º–æ–≥—É—Ç —Å–æ–¥–µ—Ä–∂–∞—Ç—å –Ω–µ—Å–∫–æ–ª—å–∫–æ —Ñ–∞–π–ª–æ–≤ —á–µ—Ä–µ–∑ –ø–µ—Ä–µ–Ω–æ—Å —Å—Ç—Ä–æ–∫–∏

# –ö–†–ò–¢–ò–ß–ù–û: –ò—Å–ø–æ–ª—å–∑—É–µ–º –≤—Ä–µ–º–µ–Ω–Ω—ã–π —Ñ–∞–π–ª –¥–ª—è –æ—Ç—Å–ª–µ–∂–∏–≤–∞–Ω–∏—è –æ—à–∏–±–æ–∫ –ø—Ä–∏ –¥–æ–±–∞–≤–ª–µ–Ω–∏–∏ —Ñ–∞–π–ª–æ–≤
ADD_ERROR_FILE=$(mktemp)
ADD_ERROR_COUNT=0

# –î–æ–±–∞–≤–ª—è–µ–º –∏–∑–Ω–∞—á–∞–ª—å–Ω–æ –∏–∑–º–µ–Ω–µ–Ω–Ω—ã–µ —Ñ–∞–π–ª—ã
echo "$CHANGED_FILES" | while read file; do
    [ ! -z "$file" ] || continue
    if git add "$file" 2>/dev/null; then
        echo "‚úì –î–æ–±–∞–≤–ª–µ–Ω: $file"
    else
        echo "‚úó –û–®–ò–ë–ö–ê: –ù–µ —É–¥–∞–ª–æ—Å—å –¥–æ–±–∞–≤–∏—Ç—å $file –≤ –∏–Ω–¥–µ–∫—Å"
        echo "ERROR" >> "$ADD_ERROR_FILE"
    fi
done

# –î–æ–±–∞–≤–ª—è–µ–º –Ω–æ–≤—ã–µ —Ñ–∞–π–ª—ã
echo "$NEW_FILES" | while read file; do
    [ ! -z "$file" ] || continue
    if [ -f "$file" ]; then
        if git add "$file" 2>/dev/null; then
            echo "‚úì –î–æ–±–∞–≤–ª–µ–Ω –Ω–æ–≤—ã–π: $file"
        else
            echo "‚úó –û–®–ò–ë–ö–ê: –ù–µ —É–¥–∞–ª–æ—Å—å –¥–æ–±–∞–≤–∏—Ç—å –Ω–æ–≤—ã–π —Ñ–∞–π–ª $file –≤ –∏–Ω–¥–µ–∫—Å"
            echo "ERROR" >> "$ADD_ERROR_FILE"
        fi
    else
        echo "‚úó –û–®–ò–ë–ö–ê: –ù–æ–≤—ã–π —Ñ–∞–π–ª $file –Ω–µ —Å—É—â–µ—Å—Ç–≤—É–µ—Ç"
        echo "ERROR" >> "$ADD_ERROR_FILE"
    fi
done

# –î–æ–±–∞–≤–ª—è–µ–º —É–¥–∞–ª–µ–Ω–Ω—ã–µ —Ñ–∞–π–ª—ã
echo "$DELETED_FILES" | while read file; do
    [ ! -z "$file" ] || continue
    if git add "$file" 2>/dev/null; then
        echo "‚úì –î–æ–±–∞–≤–ª–µ–Ω —É–¥–∞–ª–µ–Ω–Ω—ã–π: $file"
    else
        echo "‚úó –û–®–ò–ë–ö–ê: –ù–µ —É–¥–∞–ª–æ—Å—å –¥–æ–±–∞–≤–∏—Ç—å —É–¥–∞–ª–µ–Ω–Ω—ã–π —Ñ–∞–π–ª $file –≤ –∏–Ω–¥–µ–∫—Å"
        echo "ERROR" >> "$ADD_ERROR_FILE"
    fi
done

# –î–æ–±–∞–≤–ª—è–µ–º —Ñ–∞–π–ª—ã —Å –æ–±–Ω–æ–≤–ª–µ–Ω–Ω—ã–º–∏ –≤–µ—Ä—Å–∏—è–º–∏/–¥–∞—Ç–∞–º–∏ (–≤—Å–µ –∏–∑–º–µ–Ω–µ–Ω–Ω—ã–µ .py .sh .md —Ñ–∞–π–ª—ã)
# –ü–æ–ª—É—á–∞–µ–º —Å–ø–∏—Å–æ–∫ –≤—Å–µ—Ö –∏–∑–º–µ–Ω–µ–Ω–Ω—ã—Ö —Ñ–∞–π–ª–æ–≤ –ø—Ä–æ–µ–∫—Ç–∞ (–æ—Ç–Ω–æ—Å–∏—Ç–µ–ª—å–Ω–æ HEAD)
for f in *.py *.sh *.md; do
    if [ -f "$f" ]; then
        # –ü—Ä–æ–≤–µ—Ä—è–µ–º –µ—Å—Ç—å –ª–∏ –∏–∑–º–µ–Ω–µ–Ω–∏—è –≤ —Ñ–∞–π–ª–µ –æ—Ç–Ω–æ—Å–∏—Ç–µ–ª—å–Ω–æ HEAD
        if ! git diff --quiet HEAD -- "$f" 2>/dev/null; then
            # –§–∞–π–ª –∏–∑–º–µ–Ω–µ–Ω, –¥–æ–±–∞–≤–ª—è–µ–º –µ–≥–æ (git add –±–µ–∑–æ–ø–∞—Å–Ω–æ –¥–æ–±–∞–≤–ª—è–µ—Ç —Ç–æ–ª—å–∫–æ –µ—Å–ª–∏ –µ—Å—Ç—å –∏–∑–º–µ–Ω–µ–Ω–∏—è)
            if git add "$f" 2>/dev/null; then
                echo "‚úì –î–æ–±–∞–≤–ª–µ–Ω –∏–∑–º–µ–Ω–µ–Ω–Ω—ã–π: $f"
            else
                echo "‚úó –û–®–ò–ë–ö–ê: –ù–µ —É–¥–∞–ª–æ—Å—å –¥–æ–±–∞–≤–∏—Ç—å –∏–∑–º–µ–Ω–µ–Ω–Ω—ã–π —Ñ–∞–π–ª $f –≤ –∏–Ω–¥–µ–∫—Å"
                echo "ERROR" >> "$ADD_ERROR_FILE"
            fi
        fi
    fi
done

# –ö–†–ò–¢–ò–ß–ù–û: –ü—Ä–æ–≤–µ—Ä–∫–∞ –æ—à–∏–±–æ–∫ –ø—Ä–∏ –¥–æ–±–∞–≤–ª–µ–Ω–∏–∏ —Ñ–∞–π–ª–æ–≤
if [ -f "$ADD_ERROR_FILE" ] && [ -s "$ADD_ERROR_FILE" ]; then
    ADD_ERROR_COUNT=$(wc -l < "$ADD_ERROR_FILE")
    rm "$ADD_ERROR_FILE"
    stop_on_error "–ù–µ —É–¥–∞–ª–æ—Å—å –¥–æ–±–∞–≤–∏—Ç—å $ADD_ERROR_COUNT —Ñ–∞–π–ª(–æ–≤) –≤ –∏–Ω–¥–µ–∫—Å. –ü—Ä–æ–≤–µ—Ä—å—Ç–µ –ø—Ä–∞–≤–∞ –¥–æ—Å—Ç—É–ø–∞ –∏ —Å–æ—Å—Ç–æ—è–Ω–∏–µ —Ñ–∞–π–ª–æ–≤."
else
    rm -f "$ADD_ERROR_FILE"
fi

# –ü–æ–∫–∞–∑—ã–≤–∞–µ–º –∏—Ç–æ–≥–æ–≤—ã–π —Å—Ç–∞—Ç—É—Å
echo "=== –ò—Ç–æ–≥–æ–≤—ã–π —Å—Ç–∞—Ç—É—Å –∏–Ω–¥–µ–∫—Å–∞ ==="
git status --short

# –ö–†–ò–¢–ò–ß–ù–û: –ü—Ä–æ–≤–µ—Ä–∫–∞ –Ω–∞–ª–∏—á–∏—è —Ñ–∞–π–ª–æ–≤ –≤ –∏–Ω–¥–µ–∫—Å–µ
STAGED_FILES=$(git diff --cached --name-only)
if [ -z "$STAGED_FILES" ]; then
    stop_on_error "–ù–µ—Ç —Ñ–∞–π–ª–æ–≤ –≤ –∏–Ω–¥–µ–∫—Å–µ –¥–ª—è –∫–æ–º–º–∏—Ç–∞. –§–∞–π–ª—ã –Ω–µ –±—ã–ª–∏ –¥–æ–±–∞–≤–ª–µ–Ω—ã –≤ –∏–Ω–¥–µ–∫—Å."
fi

# –®–ê–ì 11: –°–æ–∑–¥–∞–µ–º –∫–æ–º–º–∏—Ç
echo "=== –®–ê–ì 11: –°–æ–∑–¥–∞–Ω–∏–µ –∫–æ–º–º–∏—Ç–∞ ==="
if ! git commit -F commit_message.txt; then
    stop_on_error "–ù–µ —É–¥–∞–ª–æ—Å—å —Å–æ–∑–¥–∞—Ç—å –∫–æ–º–º–∏—Ç. –ü—Ä–æ–≤–µ—Ä—å—Ç–µ —Å–æ–æ–±—â–µ–Ω–∏–µ –∫–æ–º–º–∏—Ç–∞ –∏ —Å–æ—Å—Ç–æ—è–Ω–∏–µ —Ä–µ–ø–æ–∑–∏—Ç–æ—Ä–∏—è."
fi

# –®–ê–ì 12: –í–µ—Ä–∏—Ñ–∏–∫–∞—Ü–∏—è –∫–æ–º–º–∏—Ç–∞
echo "=== –®–ê–ì 12: –í–µ—Ä–∏—Ñ–∏–∫–∞—Ü–∏—è –∫–æ–º–º–∏—Ç–∞ ==="
if ! git log -1 --stat > /dev/null 2>&1; then
    stop_on_error "–ù–µ —É–¥–∞–ª–æ—Å—å –ø–æ–ª—É—á–∏—Ç—å –∏–Ω—Ñ–æ—Ä–º–∞—Ü–∏—é –æ –ø–æ—Å–ª–µ–¥–Ω–µ–º –∫–æ–º–º–∏—Ç–µ. –ö–æ–º–º–∏—Ç –Ω–µ –±—ã–ª —Å–æ–∑–¥–∞–Ω."
fi
git log -1 --stat

# –ö–†–ò–¢–ò–ß–ù–û: –ü—Ä–æ–≤–µ—Ä–∫–∞ —Å–æ–∑–¥–∞–Ω–∏—è –∫–æ–º–º–∏—Ç–∞
COMMIT_HASH=$(git log -1 --format="%H")
if [ -z "$COMMIT_HASH" ]; then
    stop_on_error "–ö–æ–º–º–∏—Ç –Ω–µ –Ω–∞–π–¥–µ–Ω. –ö–æ–º–º–∏—Ç –Ω–µ –±—ã–ª —Å–æ–∑–¥–∞–Ω."
fi
echo "‚úì –ö–æ–º–º–∏—Ç —É—Å–ø–µ—à–Ω–æ —Å–æ–∑–¥–∞–Ω: $COMMIT_HASH"

# –®–ê–ì 13: –£–¥–∞–ª—è–µ–º –≤—Ä–µ–º–µ–Ω–Ω—ã–π —Ñ–∞–π–ª
echo "=== –®–ê–ì 13: –£–¥–∞–ª–µ–Ω–∏–µ –≤—Ä–µ–º–µ–Ω–Ω–æ–≥–æ —Ñ–∞–π–ª–∞ ==="
rm commit_message.txt

# –®–ê–ì 14: –ü—Ä–æ–≤–µ—Ä—è–µ–º, –Ω—É–∂–Ω–∞ –ª–∏ –ª–æ–∫–∞–ª—å–Ω–∞—è –∏—Å—Ç–æ—Ä–∏—è (–∏–∑–º–µ–Ω—ë–Ω –ª–∏ –æ–¥–∏–Ω –∏–∑ –∫–ª—é—á–µ–≤—ã—Ö —Ñ–∞–π–ª–æ–≤)
# –ö–†–ò–¢–ò–ß–ù–û: –ò—Å–ø–æ–ª—å–∑—É–µ–º –ø—Ä–æ—Å—Ç—É—é –ø—Ä–æ–≤–µ—Ä–∫—É –±–µ–∑ —Å–ª–æ–∂–Ω–æ–≥–æ regex –¥–ª—è –∏–∑–±–µ–∂–∞–Ω–∏—è –∑–∞–≤–∏—Å–∞–Ω–∏–π
echo "=== –®–ê–ì 14: –ü—Ä–æ–≤–µ—Ä–∫–∞ –Ω–µ–æ–±—Ö–æ–¥–∏–º–æ—Å—Ç–∏ –∏—Å—Ç–æ—Ä–∏–∏ ==="
NEED_HISTORY=no
if echo "$CHANGED_FILES" | grep -q "astra_automation.py"; then
    NEED_HISTORY=yes
elif echo "$CHANGED_FILES" | grep -q "astra_install.sh"; then
    NEED_HISTORY=yes
elif echo "$CHANGED_FILES" | grep -q "astra_update.sh"; then
    NEED_HISTORY=yes
fi
echo "NEED_HISTORY=$NEED_HISTORY"

# –®–ê–ì 15-16: –°–æ–∑–¥–∞–µ–º –ª–æ–∫–∞–ª—å–Ω—É—é –∏—Å—Ç–æ—Ä–∏—é (—Ç–æ–ª—å–∫–æ –µ—Å–ª–∏ NEED_HISTORY=yes)
if [ "$NEED_HISTORY" = "yes" ]; then
  echo "=== –®–ê–ì 15-16: –°–æ–∑–¥–∞–Ω–∏–µ –ª–æ–∫–∞–ª—å–Ω–æ–π –∏—Å—Ç–æ—Ä–∏–∏ ==="
  COMMIT_DATE=$(date +%Y_%m_%d_%H_%M)
  mkdir -p "History/$COMMIT_DATE"
  cp astra_automation.py "History/$COMMIT_DATE/" 2>/dev/null || true
  cp astra_install.sh "History/$COMMIT_DATE/" 2>/dev/null || true
  cp astra_update.sh "History/$COMMIT_DATE/" 2>/dev/null || true
  echo "=== –®–ê–ì 16: –í–∞–ª–∏–¥–∞—Ü–∏—è –∏—Å—Ç–æ—Ä–∏–∏ ==="
  ls -la "History/$COMMIT_DATE/"
else
  echo "–ò—Å—Ç–æ—Ä–∏—è –Ω–µ —Ç—Ä–µ–±—É–µ—Ç—Å—è (–∫–ª—é—á–µ–≤—ã–µ —Ñ–∞–π–ª—ã –Ω–µ –∏–∑–º–µ–Ω—è–ª–∏—Å—å)"
fi

# –®–ê–ì 17-18: –ü–æ–¥—Ç–≤–µ—Ä–∂–¥–µ–Ω–∏–µ –∑–∞–≤–µ—Ä—à–µ–Ω–∏—è
echo "=== –®–ê–ì 17-18: –ü–æ–¥—Ç–≤–µ—Ä–∂–¥–µ–Ω–∏–µ –∑–∞–≤–µ—Ä—à–µ–Ω–∏—è ==="
echo "–°–Ω–∏–º–æ–∫ (–∫–æ–º–º–∏—Ç) —É—Å–ø–µ—à–Ω–æ —Å–æ–∑–¥–∞–Ω!"
