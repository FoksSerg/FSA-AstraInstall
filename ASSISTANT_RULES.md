```bash
NEW_FILES=$(git status --short | grep "^??" | awk '{print $2}')
DELETED_FILES=$(git diff --name-only --diff-filter=D HEAD)
echo "=== –®–ê–ì 1: –ò–∑–º–µ–Ω–µ–Ω–Ω—ã–µ —Ñ–∞–π–ª—ã ==="
echo "–ò–∑–º–µ–Ω–µ–Ω–Ω—ã–µ —Ñ–∞–π–ª—ã: $CHANGED_FILES"
echo "–ù–æ–≤—ã–µ —Ñ–∞–π–ª—ã: $NEW_FILES"
echo "–£–¥–∞–ª–µ–Ω–Ω—ã–µ —Ñ–∞–π–ª—ã: $DELETED_FILES"

# –®–ê–ì 2: –ü—Ä–æ–≤–µ—Ä—è–µ–º —Ä–µ–∞–ª—å–Ω—ã–µ –∏–∑–º–µ–Ω–µ–Ω–∏—è —á–µ—Ä–µ–∑ git diff (–ù–ï –ø—Ä–æ –≤–µ—Ä—Å–∏–∏, –∞ –ø—Ä–æ –∫–æ–¥)
echo "=== –®–ê–ì 2: –ü—Ä–æ–≤–µ—Ä–∫–∞ —Ä–µ–∞–ª—å–Ω—ã—Ö –∏–∑–º–µ–Ω–µ–Ω–∏–π ==="
for file in $CHANGED_FILES; do
    echo "--- –ò–∑–º–µ–Ω–µ–Ω–∏—è –≤ $file ---"
    git diff HEAD "$file" | grep "^[\+\-]" | grep -v "^[\+\+\+\-\-\-]" | grep -v -E "–í–µ—Ä—Å–∏—è|Version|–¥–∞—Ç–∞|Date" | head -20
done

# –®–ê–ì 3: –°–æ–∑–¥–∞–µ–º —Ñ–∞–π–ª —Å–æ–æ–±—â–µ–Ω–∏—è –∫–æ–º–º–∏—Ç–∞ –Ω–∞ –æ—Å–Ω–æ–≤–µ —Ä–µ–∞–ª—å–Ω—ã—Ö –∏–∑–º–µ–Ω–µ–Ω–∏–π
echo "=== –®–ê–ì 3: –°–æ–∑–¥–∞–Ω–∏–µ —Ñ–∞–π–ª–∞ —Å–æ–æ–±—â–µ–Ω–∏—è –∫–æ–º–º–∏—Ç–∞ ==="
TODAY=$(date +%Y.%m.%d)
cat > commit_message.txt << COMMIT_EOF
–ó–∞–≥–æ–ª–æ–≤–æ–∫ –∫–æ–º–º–∏—Ç–∞ (–æ–ø–∏—à–∏—Ç–µ —Ä–µ–∞–ª—å–Ω—ã–µ –∏–∑–º–µ–Ω–µ–Ω–∏—è –Ω–∞ –æ—Å–Ω–æ–≤–µ git diff)

–ö—Ä–∞—Ç–∫–æ, —á—Ç–æ –∏–∑–º–µ–Ω–∏–ª–∏ (—Å–ø–∏—Å–∫–æ–º):
- –ü—É–Ω–∫—Ç 1 (–æ–ø–∏—à–∏—Ç–µ —Ä–µ–∞–ª—å–Ω—ã–µ –∏–∑–º–µ–Ω–µ–Ω–∏—è –∫–æ–¥–∞)
- –ü—É–Ω–∫—Ç 2 (–æ–ø–∏—à–∏—Ç–µ —Ä–µ–∞–ª—å–Ω—ã–µ –∏–∑–º–µ–Ω–µ–Ω–∏—è –∫–æ–¥–∞)

–û–ø–∏—Å–∞–Ω–∏–µ —É–¥–∞–ª–µ–Ω–Ω—ã—Ö –∫–ª–∞—Å—Å–æ–≤/–º–æ–¥—É–ª–µ–π:
- –û–ø–∏—Å–∞–Ω–∏–µ –µ—Å–ª–∏ —á—Ç–æ-—Ç–æ —É–¥–∞–ª–µ–Ω–æ

–ü—Ä–æ–µ–∫—Ç: FSA-AstraInstall
–î–∞—Ç–∞: $TODAY
COMMIT_EOF

# –®–ê–ì 4: –ü–æ–∫–∞–∑—ã–≤–∞–µ–º —Ñ–∞–π–ª —Å–æ–æ–±—â–µ–Ω–∏—è –∏ –∑–∞–ø—Ä–∞—à–∏–≤–∞–µ–º —Ä–∞–∑—Ä–µ—à–µ–Ω–∏–µ
echo "=== –®–ê–ì 4: –°–æ–¥–µ—Ä–∂–∏–º–æ–µ commit_message.txt ==="
cat commit_message.txt
echo ""
echo "–§–∞–π–ª commit_message.txt —Å–æ–∑–¥–∞–Ω. –û—Ç—Ä–µ–¥–∞–∫—Ç–∏—Ä—É–π—Ç–µ –µ–≥–æ –ø—Ä–∏ –Ω–µ–æ–±—Ö–æ–¥–∏–º–æ—Å—Ç–∏ –Ω–∞ –æ—Å–Ω–æ–≤–µ —Ä–µ–∞–ª—å–Ω—ã—Ö –∏–∑–º–µ–Ω–µ–Ω–∏–π –∏ –¥–∞–π—Ç–µ —Ä–∞–∑—Ä–µ—à–µ–Ω–∏–µ –ø—Ä–æ–¥–æ–ª–∂–∏—Ç—å."

# –®–ê–ì 5: –û–±–Ω–æ–≤–ª—è–µ–º –¥–∞—Ç—É —Ä–µ–ª–∏–∑–∞ –¢–û–õ–¨–ö–û –≤ –∏–∑–º–µ–Ω–µ–Ω–Ω—ã—Ö —Ñ–∞–π–ª–∞—Ö
TODAY=$(date +%Y.%m.%d)
echo "=== –®–ê–ì 5: –û–±–Ω–æ–≤–ª–µ–Ω–∏–µ –¥–∞—Ç —Ä–µ–ª–∏–∑–∞ ==="
for file in $CHANGED_FILES; do
    if [[ "$file" =~ \.(py|sh|md)$ ]]; then
        echo "–û–±–Ω–æ–≤–ª—è–µ–º –¥–∞—Ç—É —Ä–µ–ª–∏–∑–∞ –≤: $file"
        # –û–±–Ω–æ–≤–ª—è–µ–º –¥–∞—Ç—É —Ç–æ–ª—å–∫–æ –≤ —Å–∫–æ–±–∫–∞—Ö (–±–µ–∑ –∫–∞–≤—ã—á–µ–∫)
        sed -i '' "1,15s/([0-9]\{4\}\.[0-9]\{2\}\.[0-9]\{2\})/($TODAY)/g" "$file"
    fi
done

# –®–ê–ì 6: –ü—Ä–æ–≤–µ—Ä—è–µ–º –æ–±–Ω–æ–≤–ª–µ–Ω–∏–µ –¥–∞—Ç —Ä–µ–ª–∏–∑–∞
echo "=== –®–ê–ì 6: –ü—Ä–æ–≤–µ—Ä–∫–∞ –æ–±–Ω–æ–≤–ª–µ–Ω–∏—è –¥–∞—Ç —Ä–µ–ª–∏–∑–∞ ==="
for file in $CHANGED_FILES; do
    if [[ "$file" =~ \.(py|sh|md)$ ]]; then
        echo "–ü—Ä–æ–≤–µ—Ä–∫–∞ $file:"
        grep -E "($TODAY)" "$file" | head -3 || echo "–û–®–ò–ë–ö–ê: –î–∞—Ç–∞ –Ω–µ –æ–±–Ω–æ–≤–ª–µ–Ω–∞ –≤ $file"
    fi
done

# –®–ê–ì 7: –û–ø—Ä–µ–¥–µ–ª—è–µ–º —Ç–µ–∫—É—â—É—é –≤–µ—Ä—Å–∏—é –≤ —Ñ–∞–π–ª–∞—Ö
echo "=== –®–ê–ì 7: –û–ø—Ä–µ–¥–µ–ª–µ–Ω–∏–µ —Ç–µ–∫—É—â–µ–π –≤–µ—Ä—Å–∏–∏ ==="
CURRENT_VERSION=$(grep -h "V[0-9]\+\.[0-9]\+\.[0-9]\+" ASSISTANT_RULES.md astra_automation.py 2>/dev/null | grep -o "V[0-9]\+\.[0-9]\+\.[0-9]\+" | head -1)
echo "–¢–µ–∫—É—â–∞—è –≤–µ—Ä—Å–∏—è –≤ —Ñ–∞–π–ª–∞—Ö: $CURRENT_VERSION"

# –®–ê–ì 8: –û–±–Ω–æ–≤–ª—è–µ–º –≤–µ—Ä—Å–∏—é –ø—Ä–æ–µ–∫—Ç–∞ –≤–æ –í–°–ï–• —Ñ–∞–π–ª–∞—Ö
# –ß–∏—Ç–∞–µ–º –≤–µ—Ä—Å–∏—é –∏–∑ version.txt
MAJOR=$(grep "^MAJOR=" version.txt | cut -d= -f2)
MINOR=$(grep "^MINOR=" version.txt | cut -d= -f2)
CURRENT_COMMIT=$(git rev-list --count HEAD)
NEW_PATCH=$((CURRENT_COMMIT + 1))
NEW_VERSION="V${MAJOR}.${MINOR}.${NEW_PATCH}"
echo "=== –®–ê–ì 8: –û–±–Ω–æ–≤–ª–µ–Ω–∏–µ –≤–µ—Ä—Å–∏–π –ø—Ä–æ–µ–∫—Ç–∞ ==="
echo "–ù–æ–≤–∞—è –≤–µ—Ä—Å–∏—è: $NEW_VERSION (—Ç–µ–∫—É—â–∞—è: $CURRENT_VERSION)"

# –û–±–Ω–æ–≤–ª—è–µ–º –≤–µ—Ä—Å–∏—é –≤–æ –≤—Å–µ—Ö —Ñ–∞–π–ª–∞—Ö –∏—Å–ø–æ–ª—å–∑—É—è —Ç–æ—á–Ω—É—é –∑–∞–º–µ–Ω—É —Ç–µ–∫—É—â–µ–π –≤–µ—Ä—Å–∏–∏ (—Ç–æ–ª—å–∫–æ –≤ –ø–µ—Ä–≤—ã—Ö 15 —Å—Ç—Ä–æ–∫–∞—Ö)
for f in *.py *.sh *.md; do
  if [ -f "$f" ]; then
    sed -i '' "1,15s/$CURRENT_VERSION/$NEW_VERSION/g" "$f"
  fi
done

# –®–ê–ì 9: –ü—Ä–æ–≤–µ—Ä—è–µ–º –æ–±–Ω–æ–≤–ª–µ–Ω–∏–µ –≤–µ—Ä—Å–∏–π
echo "=== –®–ê–ì 9: –ü—Ä–æ–≤–µ—Ä–∫–∞ –æ–±–Ω–æ–≤–ª–µ–Ω–∏—è –≤–µ—Ä—Å–∏–π ==="
for f in *.py *.sh *.md; do
  if [ -f "$f" ]; then
    if grep -q "$NEW_VERSION" "$f"; then
        echo "‚úì –í–µ—Ä—Å–∏—è –æ–±–Ω–æ–≤–ª–µ–Ω–∞ –≤ $f"
    else
        echo "‚úó –û–®–ò–ë–ö–ê: –í–µ—Ä—Å–∏—è –ù–ï –æ–±–Ω–æ–≤–ª–µ–Ω–∞ –≤ $f"
    fi
  fi
done

# –®–ê–ì 10: –î–æ–±–∞–≤–ª—è–µ–º –∏–∑–º–µ–Ω–µ–Ω–Ω—ã–µ —Ñ–∞–π–ª—ã —è–≤–Ω–æ (–ù–ï –∏—Å–ø–æ–ª—å–∑—É–µ–º git add .)
echo "=== –®–ê–ì 10: –î–æ–±–∞–≤–ª–µ–Ω–∏–µ —Ñ–∞–π–ª–æ–≤ –≤ –∏–Ω–¥–µ–∫—Å ==="
# –î–æ–±–∞–≤–ª—è–µ–º –∏–∑–Ω–∞—á–∞–ª—å–Ω–æ –∏–∑–º–µ–Ω–µ–Ω–Ω—ã–µ —Ñ–∞–π–ª—ã
for file in $CHANGED_FILES; do
    git add "$file"
done
# –î–æ–±–∞–≤–ª—è–µ–º –Ω–æ–≤—ã–µ —Ñ–∞–π–ª—ã
for file in $NEW_FILES; do
    if [ -f "$file" ]; then
        git add "$file"
    fi
done
# –î–æ–±–∞–≤–ª—è–µ–º —É–¥–∞–ª–µ–Ω–Ω—ã–µ —Ñ–∞–π–ª—ã
for file in $DELETED_FILES; do
    git add "$file"
done
# –î–æ–±–∞–≤–ª—è–µ–º —Ñ–∞–π–ª—ã —Å –æ–±–Ω–æ–≤–ª–µ–Ω–Ω—ã–º–∏ –≤–µ—Ä—Å–∏—è–º–∏/–¥–∞—Ç–∞–º–∏ (–≤—Å–µ –∏–∑–º–µ–Ω–µ–Ω–Ω—ã–µ .py .sh .md —Ñ–∞–π–ª—ã)
# –ü–æ–ª—É—á–∞–µ–º —Å–ø–∏—Å–æ–∫ –≤—Å–µ—Ö –∏–∑–º–µ–Ω–µ–Ω–Ω—ã—Ö —Ñ–∞–π–ª–æ–≤ –ø—Ä–æ–µ–∫—Ç–∞ (–æ—Ç–Ω–æ—Å–∏—Ç–µ–ª—å–Ω–æ HEAD)
for f in *.py *.sh *.md; do
    if [ -f "$f" ]; then
        # –ü—Ä–æ–≤–µ—Ä—è–µ–º –µ—Å—Ç—å –ª–∏ –∏–∑–º–µ–Ω–µ–Ω–∏—è –≤ —Ñ–∞–π–ª–µ –æ—Ç–Ω–æ—Å–∏—Ç–µ–ª—å–Ω–æ HEAD
        if ! git diff --quiet HEAD -- "$f" 2>/dev/null; then
            # –§–∞–π–ª –∏–∑–º–µ–Ω–µ–Ω, –¥–æ–±–∞–≤–ª—è–µ–º –µ–≥–æ (git add –±–µ–∑–æ–ø–∞—Å–Ω–æ –¥–æ–±–∞–≤–ª—è–µ—Ç —Ç–æ–ª—å–∫–æ –µ—Å–ª–∏ –µ—Å—Ç—å –∏–∑–º–µ–Ω–µ–Ω–∏—è)
            git add "$f"
        fi
    fi
done
git status --short

# –®–ê–ì 11: –°–æ–∑–¥–∞–µ–º –∫–æ–º–º–∏—Ç
echo "=== –®–ê–ì 11: –°–æ–∑–¥–∞–Ω–∏–µ –∫–æ–º–º–∏—Ç–∞ ==="
git commit -F commit_message.txt

# –®–ê–ì 12: –í–µ—Ä–∏—Ñ–∏–∫–∞—Ü–∏—è –∫–æ–º–º–∏—Ç–∞
echo "=== –®–ê–ì 12: –í–µ—Ä–∏—Ñ–∏–∫–∞—Ü–∏—è –∫–æ–º–º–∏—Ç–∞ ==="
git log -1 --stat

# –®–ê–ì 13: –£–¥–∞–ª—è–µ–º –≤—Ä–µ–º–µ–Ω–Ω—ã–π —Ñ–∞–π–ª
echo "=== –®–ê–ì 13: –£–¥–∞–ª–µ–Ω–∏–µ –≤—Ä–µ–º–µ–Ω–Ω–æ–≥–æ —Ñ–∞–π–ª–∞ ==="
rm commit_message.txt

# –®–ê–ì 14: –ü—Ä–æ–≤–µ—Ä—è–µ–º, –Ω—É–∂–Ω–∞ –ª–∏ –ª–æ–∫–∞–ª—å–Ω–∞—è –∏—Å—Ç–æ—Ä–∏—è (–∏–∑–º–µ–Ω—ë–Ω –ª–∏ –æ–¥–∏–Ω –∏–∑ –∫–ª—é—á–µ–≤—ã—Ö —Ñ–∞–π–ª–æ–≤)
echo "=== –®–ê–ì 14: –ü—Ä–æ–≤–µ—Ä–∫–∞ –Ω–µ–æ–±—Ö–æ–¥–∏–º–æ—Å—Ç–∏ –∏—Å—Ç–æ—Ä–∏–∏ ==="
NEED_HISTORY=no
echo "$CHANGED_FILES" | grep -Eq '(^|/)astra_automation\.py($|)|(^|/)astra_install\.sh($|)|(^|/)astra_update\.sh($|)' && NEED_HISTORY=yes
echo "NEED_HISTORY=$NEED_HISTORY"

# –®–ê–ì 15-16: –°–æ–∑–¥–∞–µ–º –ª–æ–∫–∞–ª—å–Ω—É—é –∏—Å—Ç–æ—Ä–∏—é (—Ç–æ–ª—å–∫–æ –µ—Å–ª–∏ NEED_HISTORY=yes)
if [ "$NEED_HISTORY" = "yes" ]; then
  echo "=== –®–ê–ì 15-16: –°–æ–∑–¥–∞–Ω–∏–µ –ª–æ–∫–∞–ª—å–Ω–æ–π –∏—Å—Ç–æ—Ä–∏–∏ ==="
  COMMIT_DATE=$(date +%Y_%m_%d_%H_%M)
  mkdir -p "History/$COMMIT_DATE"
  cp astra_automation.py "History/$COMMIT_DATE/" 2>/dev/null || true
  cp astra_install.sh "History/$COMMIT_DATE/" 2>/dev/null || true
  cp astra_update.sh "History/$COMMIT_DATE/" 2>/dev/null || true
  echo "=== –®–ê–ì 16: –í–∞–ª–∏–¥–∞—Ü–∏—è –∏—Å—Ç–æ—Ä–∏–∏ ==="
  ls -la "History/$COMMIT_DATE/"
else
  echo "–ò—Å—Ç–æ—Ä–∏—è –Ω–µ —Ç—Ä–µ–±—É–µ—Ç—Å—è (–∫–ª—é—á–µ–≤—ã–µ —Ñ–∞–π–ª—ã –Ω–µ –∏–∑–º–µ–Ω—è–ª–∏—Å—å)"
fi

# –®–ê–ì 17-18: –ü–æ–¥—Ç–≤–µ—Ä–∂–¥–µ–Ω–∏–µ –∑–∞–≤–µ—Ä—à–µ–Ω–∏—è
echo "=== –®–ê–ì 17-18: –ü–æ–¥—Ç–≤–µ—Ä–∂–¥–µ–Ω–∏–µ –∑–∞–≤–µ—Ä—à–µ–Ω–∏—è ==="
echo "–°–Ω–∏–º–æ–∫ (–∫–æ–º–º–∏—Ç) —É—Å–ø–µ—à–Ω–æ —Å–æ–∑–¥–∞–Ω!"
```
